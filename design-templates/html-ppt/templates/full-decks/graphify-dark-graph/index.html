04 / 08

One command

From zero to graph in about 90 seconds

$ graphify ~/notes --out ./graph

# graphify.config.yaml
ingest:
  paths: [~/notes, ~/code/docs]
  include: ["*.md", "*.pdf", "*.py"]

extract:
  model: claude-opus-4-6
  schema: [concept, person, event, relation]

render:
  engine: d3-force
  audit: true     # every edge includes a source span
